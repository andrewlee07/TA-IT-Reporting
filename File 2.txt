// ═══════════════════════════════════════════════════════════════════
// HELPERS
// ═══════════════════════════════════════════════════════════════════
function svcClass(pct, target) {
  const p = pctNum(pct), t = pctNum(target);
  return p >= t ? 'up' : p >= t - 1 ? 'warn' : 'down';
}
function svcPctClass(pct, target) {
  const p = pctNum(pct), t = pctNum(target);
  return p >= t ? 'full' : p >= t - 1 ? 'warn' : 'down';
}
function svcStatusHTML(pct, target) {
  const p = pctNum(pct), t = pctNum(target);
  if (p >= t)    return '<span class="svc-status pill-green">Above target</span>';
  if (p >= t-1)  return '<span class="svc-status pill-amber">Within tolerance</span>';
  return '<span class="svc-status pill-red">Below target</span>';
}
function ragColor(r){ return r==='Green'?'green':r==='Amber'?'amber':'red'; }
